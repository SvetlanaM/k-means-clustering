class kStredu:
    #data -----------------------------------------------------------------------------------------
    X = []  #proměnná pro uložení dat jako numpy array
    stred = [] #proměnná pro uložení středů shluků, v průběhu se mění
    prislusnost = [] #proměnná pro uložení příslušnosti pozorování ke shlukům
    hodnotaUceloveFunkce = float("Inf") #hodnota účelové funkce
    k = 0 #počet shluků
    pIteraci = 0 #počet iterací
    
    #při inicializace vložím uložím do proměnné k počet shluků ------------------------------------
    def __init__(self,k):
        #Úloha 1 - Předání dat #################################
        #zde vložte kód
        self.k = k
        ########################################################
     
    #vložení dat ----------------------------------------------------------------------------------
    def vlozData(self,X):
        #Úloha 1 - Předání dat #################################
        #zde vložte kód
        self.X = X
        ########################################################
    
        self.prislusnot = X.shape[0]*[0]
    
    #inicializace středů --------------------------------------------------------------------------
    def inicializujStredy(self,cislaPozorovani):
        self.stred=self.X[cislaPozorovani,:]
        
    #přiřazení - E-step ---------------------------------------------------------------------------
    def prirazeni(self):
        #inicializace pole pro uložení vzdáleností
        vzdalenost = np.array([[0.0 for col in range(self.k)] for row in range(self.X.shape[0])])
        
        #Úloha 4 - přiřazení ###################################
        #zde vložte kód
        #procházím shluky
        for i in range(0,self.k):
            rozdil = self.X-np.array([self.stred[i,:]]*self.X.shape[0])
            vzdalenost[:,i] = list([np.linalg.norm(x) for x in rozdil]) 
        ######################################################## 
        
        #uložím si příslušnost ke shlukům
        self.prislusnost = vzdalenost.argmin(axis=1)
        
        #přidám hodnotu účelové funkce
        self.hodnotaUceloveFunkce = sum(vzdalenost.min(axis=1))
    
    #optimalizace - M-step ------------------------------------------------------------------------
    def optimalizace(self):
        #Úloha 5 - optimalizace ###############################
        for i in range(0,self.k):
            self.stred[i,:]=self.X[self.prislusnost == i,:].mean(axis=0)
        #######################################################
        
        #inkrementace počtu iterací
        self.pIteraci+=1
            
    #vykreslení dat -------------------------------------------------------------------------------
    def vykresliData(self):
        #Úloha 2 - Vykreslení dat ##############################
        #zde vložte kód
        plt.scatter(self.X[:,0],self.X[:,1])
        ########################################################
        
        #vynucené vykreslení
        plt.show()
        
    #vykreslení shluků ----------------------------------------------------------------------------
    def vykresliShlukovani(self):
        #Úloha 3 - Vykreslení shluků ###########################
        #zde vložte kód
        plt.scatter(self.X[:,0],self.X[:,1],c=self.prislusnost,cmap='rainbow')
        plt.scatter(self.stred[:, 0], self.stred[:, 1], s = 300, c = "black",edgecolors="black",marker="+")
        ########################################################
        
        #vynucené vykreslení
        plt.show()
        
    #hodnota účelové funkce ----------------------------------------------------------------------
    def ucelovaFunkce(self):
        print(self.hodnotaUceloveFunkce)

Zajímavější data¶

Data jsou pro vás vytvořena v proměnné data a zobrazena. Udělejte:

  • Vytvořte třídu mojeShlukovani.
  • Nastavte počet shluků, které chcete najít.
  • Nahrajte do ní data.
  • Vyberte si nějaké počáteční body jako startovní (MUSÍ JICH BÝT STEJNÝ POČET JAKO POČET SHLUKŮ k).
  • Proveďte několik iterací.

Pohrajte se s tím, zkuste různé varianty.
