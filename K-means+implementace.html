class kStredu:
    #data -----------------------------------------------------------------------------------------
    X = []  #promnn pro uloen dat jako numpy array
    stred = [] #promnn pro uloen sted shluk, v prbhu se mn
    prislusnost = [] #promnn pro uloen pslunosti pozorovn ke shlukm
    hodnotaUceloveFunkce = float("Inf") #hodnota elov funkce
    k = 0 #poet shluk
    pIteraci = 0 #poet iterac
    
    #pi inicializace vlom ulom do promnn k poet shluk ------------------------------------
    def __init__(self,k):
        #loha 1 - Pedn dat #################################
        #zde vlote kd
        self.k = k
        ########################################################
     
    #vloen dat ----------------------------------------------------------------------------------
    def vlozData(self,X):
        #loha 1 - Pedn dat #################################
        #zde vlote kd
        self.X = X
        ########################################################
    
        self.prislusnot = X.shape[0]*[0]
    
    #inicializace sted --------------------------------------------------------------------------
    def inicializujStredy(self,cislaPozorovani):
        self.stred=self.X[cislaPozorovani,:]
        
    #piazen - E-step ---------------------------------------------------------------------------
    def prirazeni(self):
        #inicializace pole pro uloen vzdlenost
        vzdalenost = np.array([[0.0 for col in range(self.k)] for row in range(self.X.shape[0])])
        
        #loha 4 - piazen ###################################
        #zde vlote kd
        #prochzm shluky
        for i in range(0,self.k):
            rozdil = self.X-np.array([self.stred[i,:]]*self.X.shape[0])
            vzdalenost[:,i] = list([np.linalg.norm(x) for x in rozdil]) 
        ######################################################## 
        
        #ulom si pslunost ke shlukm
        self.prislusnost = vzdalenost.argmin(axis=1)
        
        #pidm hodnotu elov funkce
        self.hodnotaUceloveFunkce = sum(vzdalenost.min(axis=1))
    
    #optimalizace - M-step ------------------------------------------------------------------------
    def optimalizace(self):
        #loha 5 - optimalizace ###############################
        for i in range(0,self.k):
            self.stred[i,:]=self.X[self.prislusnost == i,:].mean(axis=0)
        #######################################################
        
        #inkrementace potu iterac
        self.pIteraci+=1
            
    #vykreslen dat -------------------------------------------------------------------------------
    def vykresliData(self):
        #loha 2 - Vykreslen dat ##############################
        #zde vlote kd
        plt.scatter(self.X[:,0],self.X[:,1])
        ########################################################
        
        #vynucen vykreslen
        plt.show()
        
    #vykreslen shluk ----------------------------------------------------------------------------
    def vykresliShlukovani(self):
        #loha 3 - Vykreslen shluk ###########################
        #zde vlote kd
        plt.scatter(self.X[:,0],self.X[:,1],c=self.prislusnost,cmap='rainbow')
        plt.scatter(self.stred[:, 0], self.stred[:, 1], s = 300, c = "black",edgecolors="black",marker="+")
        ########################################################
        
        #vynucen vykreslen
        plt.show()
        
    #hodnota elov funkce ----------------------------------------------------------------------
    def ucelovaFunkce(self):
        print(self.hodnotaUceloveFunkce)
